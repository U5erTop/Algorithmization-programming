            Console.WriteLine("\n--- 12. –°–∫–æ–ª—å–∫–æ —Ç–æ–≤–∞—Ä–æ–≤ —ç–ª–µ–∫—Ç—Ä–æ–Ω–∏–∫–∏? ---");

            var mostExpensiveElectronics = products

            var groupedByCategory = products

            foreach (var group in groupedByCategory)
            {
                Console.WriteLine($"\n{group.Key}:");

            }
